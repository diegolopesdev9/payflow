const calculateMonthlyExpenses = () => {
  if (!bills || bills.length === 0) return Array(30).fill(0);

  const today = new Date();
  const expenses = Array(30).fill(0);

  bills.forEach((bill: any) => {
    if (!bill.dueDate || !bill.amount) return;

    const billDate = new Date(bill.dueDate);
    const diffTime = today.getTime() - billDate.getTime();
    const diffDays = Math.floor(diffTime / (1000 * 60 * 60 * 24));

    if (diffDays >= 0 && diffDays < 30) {
      const index = 29 - diffDays; // Inverte para mostrar do mais antigo ao mais recente
      expenses[index] += bill.amount / 100;
    }
  });

  // Agrupa em 6 barras (cada uma representa ~5 dias)
  const grouped = [];
  for (let i = 0; i < 6; i++) {
    const start = i * 5;
    const end = start + 5;
    const sum = expenses.slice(start, end).reduce((acc, val) => acc + val, 0);
    grouped.push(sum);
  }

  return grouped;
};
